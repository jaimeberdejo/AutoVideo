         script: ScriptOutput = workdir.read_checkpoint("script", ScriptOutput)

       Cache per-slide narration edits in session_state:

                 height=120,
             )
             if edited != st.session_state["scr_edited_narrations"].get(idx, slide.narration):

             col_save, _ = st.columns([1, 3])

                 if st.button(f"Guardar edición slide {idx + 1}", key=f"save_{idx}"):

                     st.success("Guion guardado. Etapas posteriores invalidadas.")
