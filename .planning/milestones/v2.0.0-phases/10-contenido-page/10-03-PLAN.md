Imports (all at module top, no lazy streamlit import needed — this file is only ever imported by app.py which already runs inside Streamlit):
```python
from __future__ import annotations
import streamlit as st
import yaml
from avideo.models.bullets import BulletsInput
from avideo.stages.bullets_gen import generate_bullets, validate_duration, DURATION_MIN, DURATION_MAX
from avideo.utils.workdir import WorkdirManager
```

render(workdir: WorkdirManager) -> bool implementation:

SECTION 1 — Page header:
  st.subheader("Fase 1 — Contenido")
  st.caption("Define el tema, duración y los bullets de tu presentación.")

SECTION 2 — Topic + Duration inputs:
  Use two columns: col1, col2 = st.columns([3, 1])

  col1: topic = st.text_input("Tema de la presentación", key="cnt_topic",
                               placeholder="p. ej. Introducción a la inteligencia artificial")

  col2: duration_raw = st.number_input("Duración objetivo (s)", key="cnt_duration",
                                        min_value=DURATION_MIN, max_value=DURATION_MAX,
                                        value=120, step=15)

  Duration is always valid because min_value/max_value enforce bounds in the widget.
  But call validate_duration(duration_raw) anyway for defense — catch ValueError and
  show st.error() + return False immediately if raised.

SECTION 3 — Source choice (CNT-02):
  source = st.radio(
      "¿Cómo quieres crear los bullets?",
      options=["Escribir mis bullets", "Generar desde el tema (Claude)"],
      key="cnt_source",
      horizontal=True,
  )

SECTION 4 — Bullet generation spinner (only when source == "Generar..."):
  Use a session_state flag "cnt_generated_bullets" (list[str] | None) to store the
  last generated set so it survives reruns without re-calling the API.

  if source == "Generar desde el tema (Claude)":
      if st.button("Generar bullets", key="cnt_gen_btn", disabled=not bool(topic)):
          if not topic.strip():
              st.warning("Escribe un tema antes de generar.")
          else:
              with st.spinner("Generando bullets con Claude..."):
                  try:
                      bullets_raw = generate_bullets(topic.strip(), int(duration_raw))
                      st.session_state["cnt_generated_bullets"] = bullets_raw
                  except Exception as e:
                      st.error(f"Error al generar bullets: {e}")

SECTION 5 — Editor (BOTH paths use st.data_editor, CNT-03):
  Determine initial data:
  - "Escribir mis bullets": start with session_state.get("cnt_manual_bullets") or [{"bullet": ""}]
  - "Generar desde el tema": use session_state.get("cnt_generated_bullets") or []

  Build a list of dicts for the editor:
  if source == "Generar desde el tema (Claude)":
      raw_list = st.session_state.get("cnt_generated_bullets") or []
  else:
      raw_list = st.session_state.get("cnt_manual_bullets") or [""]

  editor_data = [{"bullet": b} for b in raw_list] if raw_list else [{"bullet": ""}]

  edited = st.data_editor(
      editor_data,
      num_rows="dynamic",
      column_config={"bullet": st.column_config.TextColumn("Bullet", width="large")},
      use_container_width=True,
      key="cnt_bullets_editor",
  )

  # Persist manual bullets back to session_state so they survive reruns
  if source == "Escribir mis bullets":
      st.session_state["cnt_manual_bullets"] = [r["bullet"] for r in edited if r.get("bullet")]

  approved_bullets = [r["bullet"].strip() for r in edited if r.get("bullet", "").strip()]

SECTION 6 — Gate condition + Aprobar button:
  gate_met = False
  can_approve = bool(topic.strip()) and len(approved_bullets) >= 1

  if not topic.strip():
      st.info("Escribe un tema para continuar.")
  elif len(approved_bullets) < 1:
      st.info("Añade al menos un bullet para continuar.")
  else:
      # Show a preview caption
      st.caption(f"{len(approved_bullets)} bullet(s) listos para aprobar.")

      if st.button("Aprobar bullets y continuar", key="cnt_approve_btn", type="primary",
                   disabled=not can_approve):
          # Build BulletsInput and persist to workdir
          bi = BulletsInput(title=topic.strip(), bullets=approved_bullets)
          # Write workdir/bullets.yaml (same format as avideo generate --bullets consumes)
          bullets_yaml_path = workdir.root / "bullets.yaml"
          bullets_yaml_path.write_text(
              yaml.safe_dump(bi.model_dump(), allow_unicode=True, sort_keys=False),
              encoding="utf-8",
          )
          # Write checkpoint and mark phase 1 done
          workdir.write_checkpoint("context", bi)
          workdir.mark_done("context")
          # Update run_config in session_state for downstream phases
          rc = st.session_state.get("run_config", {})
          rc["topic"] = topic.strip()
          rc["duration"] = int(duration_raw)
          st.session_state["run_config"] = rc
          gate_met = True
          st.success("Bullets aprobados. Puedes continuar al guion.")

  # If context is already done (resume from workdir), gate is met
  if workdir.is_done("context"):
      gate_met = True

  return gate_met

--- END SPEC ---
