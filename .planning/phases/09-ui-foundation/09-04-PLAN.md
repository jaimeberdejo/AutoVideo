**Sidebar:**
```python
with st.sidebar:
    st.title("Studio Guiado")
    st.caption(f"📁 {st.session_state['workdir_path']}")
    st.divider()
    for phase_num, phase_name in PHASES:
        current = st.session_state["phase"]
        completion_stage = PHASE_COMPLETION_STAGE[phase_num]
        if workdir.is_done(completion_stage):
            marker = "✅"
        elif phase_num == current:
            marker = "▶"
        elif phase_num < current:
            marker = "✅"  # completed but workdir marker may be missing in placeholder
        else:
            marker = "○"
        st.write(f"{marker} Fase {phase_num}: {phase_name}")
    st.divider()
    if st.button("Nuevo proyecto", key="new_project"):
        # Clear session state and workdir to start fresh
        for key in ["phase", "workdir_path", "run_config"]:
            if key in st.session_state:
                del st.session_state[key]
        st.rerun()
```

**Navigation footer (sticky bottom via st.container):**
```python
st.divider()
col_back, col_spacer, col_next = st.columns([1, 4, 1])

with col_back:
    back_disabled = (current_phase == 1)
    if st.button("← Atrás", disabled=back_disabled, key="nav_back"):
        st.session_state["_confirm_back"] = True

# Back-navigation confirmation dialog
if st.session_state.get("_confirm_back"):
    with st.container(border=True):
        completion_stage_for_current = PHASE_COMPLETION_STAGE.get(current_phase - 1, "context")
        st.warning(
            f"⚠ Volver a la Fase {current_phase - 1} invalidará el trabajo posterior "
            f"(a partir de la etapa '{completion_stage_for_current}'). ¿Continuar?"
        )
        c1, c2 = st.columns(2)
        if c1.button("Sí, volver atrás", key="confirm_back_yes"):
            # Invalidate downstream from the completion stage of the phase we're going BACK TO
            from_stage = PHASE_COMPLETION_STAGE.get(current_phase - 1, "context")
            workdir.invalidate_downstream(from_stage)
            st.session_state["phase"] = max(1, current_phase - 1)
            st.session_state["_confirm_back"] = False
            st.rerun()
        if c2.button("Cancelar", key="confirm_back_no"):
            st.session_state["_confirm_back"] = False
            st.rerun()

with col_next:
    if st.button(
        "Aprobar y continuar →",
        disabled=not gate_met,
        type="primary",
        key="nav_next",
    ):
        advance_phase()
```

    """Render Phase N placeholder body.
